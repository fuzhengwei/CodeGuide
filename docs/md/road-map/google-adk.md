    public ChatResponse chat(@RequestBody ChatRequest req) {

        }
        log.info("ä½¿ç”¨ä¼šè¯ID:{}", sessionId);
        List<String> outputs = agentService.chat(req.getUserId(), sessionId, req.getMessage());
